Python SDK Usage
from redvox.cloud.config import RedVoxConfig
from redvox.cloud.client import CloudClient

# Connect to RedVox cloud
config = RedVoxConfig(
    username="your@email.com",
    password="your_password",
    protocol="https",
    host="redvox.io",
    port=8080,
)
client = CloudClient(config)

# Authenticate
auth = client.authenticate_user("your@email.com", "your_password")
if auth.is_success():
    print(f"Token: {auth.auth_token}")

# Request station stats
start_ts_s = 1608508800
end_ts_s = 1608512400
station_ids = ["1637681016"]
stats = client.request_station_stats(start_ts_s, end_ts_s, station_ids)

# Request data range (for time-based queries)
# See redvox.cloud.data_client for DataRangeReq usage

client.close()
